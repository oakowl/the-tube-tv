/** * ==================================================================== * 1. DATA MODEL: The Single Source of Truth (AppState) * All application state is encapsulated here. * ==================================================================== */ let AppState = { // Simulation State currentChannel: 1, isPlaying: false, currentTime: 0, // Time in seconds (resets on new segment) // Content State contentType: 'INFO', // Possible values: 'INFO', 'PROGRAM', 'AD', 'BREAK' // Master data array (Simulated data structure) contentData: [ // Example content segment structure { id: 1, type: 'INFO', title: 'Welcome Message', duration: 10, programName: 'Local Broadcast', adContent: 'Welcome!', nextContentId: 2 }, { id: 2, type: 'PROGRAM', title: 'The History Show', duration: 30, programName: 'History Tonight', adContent: null, nextContentId: 3 }, { id: 3, type: 'AD', title: 'XYZ Corp Commercial', duration: 15, programName: null, adContent: 'Buy XYZ!', nextContentId: 4 }, { id: 4, type: 'BREAK', title: 'Network Break', duration: 5, programName: null, adContent: '...', nextContentId: 5 }, { id: 5, type: 'PROGRAM', title: 'Local News Report', duration: 60, programName: 'Evening News', adContent: null, nextContentId: null } ], // Tracks the currently active segment ID currentSegmentId: 1 }; /** * ==================================================================== * 2. THE MANAGER (Controller): Handles all state transitions and logic. * ==================================================================== */ class TVManager { constructor() { this.setupInitialState(); this.bindEventListeners(); this.startSimulationLoop(); } // --- Initialization and Setup --- setupInitialState() { // Initial state setup logic AppState.currentSegmentId = 1; this.updateDisplay(); // Initial render } /** * @private * Simulates the passage of time and triggers the next state. * Must be called whenever the state should progress. */ advanceTime(deltaTime) { // Increment time only if playing if (!AppState.isPlaying) return; try { AppState.currentTime += deltaTime; const currentSegment = AppState.contentData.find(c => c.id === AppState.currentSegmentId); // Check if the segment time has elapsed if (AppState.currentTime >= currentSegment.duration) { this.transitionToNextSegment(); } else { // If not time to transition, just update the clock this.updateDisplay(); } } catch (error) { console.error("Time advancement error:", error); this.handleError("Failed to advance time."); } } // --- Core Transition Logic --- /** * Moves the state machine to the next content segment. * This is the core method for scheduling changes. */ transitionToNextSegment() { const currentSegment = AppState.contentData.find(c => c.id === AppState.currentSegmentId); if (!currentSegment || !currentSegment.nextContentId) { console.warn("End of broadcast reached."); AppState.isPlaying = false; AppState.currentTime = 0; this.updateDisplay(); return; } // 1. Update State: Set the new segment ID AppState.currentSegmentId = currentSegment.nextContentId; // 2. Find the new segment data const nextSegment = AppState.contentData.find(c => c.id === AppState.currentSegmentId); if (nextSegment) { // 3. Reset Time: Start the new segment from zero AppState.currentTime = 0; // 4. Update and Render this.updateDisplay(); } else { // Handle case where next ID points to non-existent content this.handleError(`Critical content data missing for ID ${AppState.currentSegmentId}`); } } // --- Public User Actions (Event Handlers) --- /** Handles a user pressing 'Next' button. */ handleNextClick() { if (AppState.isPlaying) { // If running, clicking next might be redundant, but we allow it. this.transitionToNextSegment(); } else { // If stopped, manually force the transition this.transitionToNextSegment(); } } /** Handles a user pressing 'Previous' button (e.g., Skip Ad). */ handlePrevClick() { // A complex feature. For simplicity, we just force a skip backward. if (AppState.contentData.find(c => c.id === AppState.currentSegmentId)?.type === 'AD') { this.transitionToNextSegment(); } else { this.handleError("Cannot skip backward from the current segment type."); } } // --- Simulation Loop --- /** Starts the time-based simulation loop. */ startSimulationLoop() { if (this.simulationInterval) { clearInterval(this.simulationInterval); // Ensure only one running } this.simulationInterval = setInterval(() => { if (AppState.isPlaying) { this.advanceTime(1); // Advance by 1 second } }, 1000); // Update every 1 second } // --- Robustness and Error Handling --- /** Global handler for catastrophic failure points. */ handleError(message, event = null) { console.error("--- APPLICATION ERROR ---", message, event); // Display a generic error message on the TV screen document.getElementById('adContent').innerHTML = `
📡 Signal Lost: ${message}
`; AppState.isPlaying = false; } // --- Event Binding --- bindEventListeners() { // Using delegated events or direct assignment is best practice document.getElementById('nextBtn').addEventListener('click', () => this.handleNextClick()); document.getElementById('prevBtn').addEventListener('click', () => this.handlePrevClick()); } } /** * ==================================================================== * 3. THE VIEW (DOM Manipulator): Reads state and renders the display. * ==================================================================== */ /** * This function is purely responsible for reading the AppState and updating the DOM. * It should contain NO business logic. */ function updateDisplay() { const state = AppState; const segment = state.contentData.find(c => c.id === state.currentSegmentId); if (!segment) { // Should have been caught by transitionToNextSegment, but good fail-safe return; } // --- Update Basic Info Bars --- document.getElementById('channelDisplay').textContent = `Ch ${state.currentChannel}`; document.getElementById('currentTimeDisplay').textContent = `${Math.floor(state.currentTime)}s/${segment.duration}s`; // --- Update Main Program Info --- document.getElementById('programName').textContent = segment.programName || 'N/A'; document.getElementById('programTitle').textContent = segment.title; // --- Update Main Content Area (Ad or Info) --- let adHtml = ''; if (segment.type === 'AD' && segment.adContent) { adHtml = `

⚠️ ADVERTISEMENT! ⚠️

${segment.adContent}

`; } else if (segment.type === 'INFO') { adHtml = `

Welcome Viewer!

This is an introductory message for ${segment.programName}.

`; } else { adHtml = `

Live Broadcast Feed

${segment.adContent || 'Program running smoothly...'}

`; } document.getElementById('adContent').innerHTML = adHtml; // Accessibility improvement: Announce the state change const statusAnnouncement = document.getElementById('statusAnnouncement'); if (statusAnnouncement) { statusAnnouncement.setAttribute('aria-live', 'polite'); statusAnnouncement.textContent = `Channel ${state.currentChannel}. Showing ${segment.type}: ${segment.title}. Time remaining: ${segment.duration - Math.floor(state.currentTime)} seconds.`; } } /** * ==================================================================== * 4. INITIALIZATION * ==================================================================== * (This code runs when the DOM is loaded) */ document.addEventListener('DOMContentLoaded', () => { // Instantiate the manager, which sets up the loop and initial state. window.tvManager = new TVManager(); });
